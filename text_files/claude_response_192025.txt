```bash
find . -name "safe_browser_manager.py" -type f
```

```bash
ls -la agents/
```

```bash
# プロジェクト全体でbrowser関連のファイルを検索

```

```bash
# safe_browser_manager.py が見つかった場合、内容を確認
if [ -f "safe_browser_manager.py" ]; then
    echo "=== safe_browser_manager.py の内容 ==="
    cat safe_browser_manager.py
elif [ -f "agents/safe_browser_manager.py" ]; then
    echo "=== agents/safe_browser_manager.py の内容 ==="
```
